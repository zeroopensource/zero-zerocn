Patched Sidebar Component.

- Added `openMobile` prop, mobile sidebar does not use `open` prop
- Fix for mobile sidebar not closing upon routing<br />
  https://github.com/shadcn-ui/ui/issues/5561
- Fix for `--sidebar-width-mobile`<br />
  https://github.com/shadcn-ui/ui/issues/5509

```tsx
<SidebarProvider
  className="flex flex-col"
  open={isSidebarEnabled ? undefined : false}
  openMobile={isSidebarEnabled ? undefined : false}
  style={
    {
      "--sidebar-width": "12rem",
      "--sidebar-width-mobile": "12rem",
    } as CSSProperties
  }
>
```
```tsx
export function AppSidebar({ ...props }: React.ComponentProps<typeof Sidebar>) {
  const { setOpenMobile } = useSidebar();
  const pathname = usePathname();
  useEffect(() => {
    setOpenMobile(false);
  }, [pathname, setOpenMobile]);
  return (
    <Sidebar {...props} >
      '...'
    </Sidebar>
  );
}
```

### Usage

Install package

```bash
pnpm dlx shadcn@latest add https://zerocn.zeroopensource.org/r/sidebar.json
```

Install package ([workspace](/docs/workspace))

```bash
pnpm --filter=$WORKSPACE shadcn add https://zerocn.zeroopensource.org/r/sidebar.json
```
